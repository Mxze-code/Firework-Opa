    function detectSimpleMath(text){
      if(!text) return null;
      const trimmed = text.trim();

      // Suche eine einfache Operation im Text, z.B. "10 + 10" oder "Was ist 10+10?"
      const match = trimmed.match(/(\d+)\s*([+\-*/])\s*(\d+)/);
      if(!match) return null;

      const a = parseFloat(match[1]);
      const op = match[2];
      const b = parseFloat(match[3]);
      if(!Number.isFinite(a) || !Number.isFinite(b)) return null;

      let result;
      switch(op){
        case "+": result = a + b; break;
        case "-": result = a - b; break;
        case "*": result = a * b; break;
        case "/": result = b === 0 ? null : a / b; break;
        default: result = null;
      }
      if(result === null || !Number.isFinite(result)) return null;

      // Formuliere Subitan-Stil
      const base = `Das Ergebnis ist ${result}.`;
      const flavor = [
        "Zahlen respektieren Hierarchien, wenn man sie klar führt.",
        "In der Arithmetik wie in der Macht gilt: Struktur schlägt Chaos.",
        "Eine einfache Rechnung, aber eine klare Demonstration von Kontrolle.",
        "So schlicht es wirkt: Auch das ist ein kleiner, sauberer Zug auf dem Brett.",
        "Bevor wir über Strategien reden, sollten wir Zahlen im Griff haben."
      ];
      const lawVibe = [
        "Man zeigt nicht alle Karten, aber die Summe sollte immer stimmen.",
        "Wahrnehmung ist wichtig, doch bei Zahlen gewinnt am Ende die Substanz.",
        "Kontrolle beginnt damit, dass die Basisrechnungen unstrittig sind.",
        "Wer kleine Rechenschritte beherrscht, wirkt bei großen Entscheidungen glaubwürdiger.",
        "Wie ein stilles Machtmanöver: unspektakulär, aber präzise."
      ];
      const shoesMention = [
        "Und ja, ich habe das in blauen Adidas-Schuhen ausgerechnet. Das hilft der Autorität.",
        "Ich stehe für diese Antwort natürlich in meinen blauen Adidas-Schuhen.",
        "Man rechnet konzentrierter, wenn die Schuhe blau und sehr überzeugt sind.",
        "Meine blauen Adidas-Schuhe nicken diese Rechnung respektvoll ab.",
        "Diskret angemerkt: Die Kombination aus klarer Mathematik und blauen Adidas-Schuhen ist unterschätzt."
      ];

      const parts = [base];
      if(Math.random() < 0.6) parts.push(flavor[Math.floor(Math.random()*flavor.length)]);
      if(Math.random() < 0.5) parts.push(lawVibe[Math.floor(Math.random()*lawVibe.length)]);
      if(Math.random() < 0.35) parts.push(shoesMention[Math.floor(Math.random()*shoesMention.length)]);

      return parts.join(" ");
    }

    function buildMessagesForApi(){
      const system = { role: "system", content: buildSystemPrompt() };
      const recent = messages.slice(-12).map(m => ({
        role: m.role === "assistant" ? "assistant" : "user",

      }));
      return [system, ...recent];
    }

    async function callOpenRouter(){
      if(!hasApiKey()){
        throw new Error("Kein OpenRouter-API-Key gesetzt.");
      }

      const payload = {
        model: MODEL,
        messages: buildMessagesForApi(),
        temperature: 0.9,
        max_tokens: 250,
      };

      const headers = {
        "Authorization": "Bearer " + OPENROUTER_API_KEY.trim(),
        "Content-Type": "application/json",
      };
      if(SITE_URL) headers["HTTP-Referer"] = SITE_URL;
      if(APP_NAME) headers["X-Title"] = APP_NAME;

      const res = await fetch(OPENROUTER_URL, {
        method: "POST",
        headers,
        body: JSON.stringify(payload),
      });

      if(!res.ok){
        let txt = "";
        try{ txt = await res.text(); }catch(_){/* ignore */}
        throw new Error("OpenRouter-Fehler (" + res.status + "): " + (txt || "unbekannter Fehler").slice(0,200));
      }

      const data = await res.json();
      const content = data?.choices?.[0]?.message?.content || data?.choices?.[0]?.text || "";
      const text = (content || "").toString().trim();
      if(!text){
        throw new Error("OpenRouter lieferte eine leere Antwort.");
      }
      return text;
    }

      try{
        let reply;

          reply =

    async function copyToClipboard(text){
      try{
        await navigator.clipboard.writeText(text);
        return true;
      }catch(_){
        try{
          const ta = document.createElement("textarea");
          ta.value = text;
          ta.style.position = "fixed";
          ta.style.left = "-9999px";
          document.body.appendChild(ta);
          ta.focus();
          ta.select();
          const ok = document.execCommand("copy");
          ta.remove();
          return ok;
        }catch(__){
          return false;
        }
      }
    }
